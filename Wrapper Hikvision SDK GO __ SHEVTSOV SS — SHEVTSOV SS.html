Invólucro Hikvision SDK GO

Procure uma solução para ir para o hikvision sdk.

Projeto de solução:

mkdir htest  
cd htest 
go mod init example.com/htest

Sugestão:

package main

import (
        "log"
        "github.com/sarjsheff/hiklib"
)

func main() {
        log.Println(hiklib.HikVersion())
}

Baixe o Hikvision Device Network SDK (para Linux de 64 bits) em https://www.hikvision.com/en/support/download/sdk/ e instale-o em (EN-HCNetSDKV6.1.6.3_build20200925_Linux64) para docker

Запускаем:

docker run --platform linux/amd64 -it --rm  -v $(pwd)/EN-HCNetSDKV6.1.6.3_build20200925_Linux64:/hiksdk -v $(pwd):/app -w /app -e LD_LIBRARY_PATH="/hiksdk/lib" -e CGO_CXXFLAGS="-I/hiksdk/incEn/" -e CGO_LDFLAGS="-L/hiksdk/lib -lhcnetsdk" golang:1.16.6 go run .

Собираем:

docker run --platform linux/amd64 -it --rm  -v $(pwd)/EN-HCNetSDKV6.1.6.3_build20200925_Linux64:/hiksdk -v $(pwd):/app -w /app -e LD_LIBRARY_PATH="/hiksdk/lib" -e CGO_CXXFLAGS="-I/hiksdk/incEn/" -e CGO_LDFLAGS="-L/hiksdk/lib -lhcnetsdk" golang:1.16.6 go build

Baixar vídeos da câmera:

package main

import (
	"errors"
	"fmt"
	"log"
	"os"

	"github.com/sarjsheff/hiklib"
)

func main() {
	log.Println(hiklib.HikVersion())
	u, _ := hiklib.HikLogin("192.168.1.111", "user", "password")
	_, v := hiklib.HikListVideo(u)
	log.Printf("Video count: %d\n", v.Count)
	os.Mkdir("tmp", 0600)
	for i := 0; i < v.Count; i++ {
		tofile := fmt.Sprintf("./tmp/%s.mpeg", v.Videos[i].Filename)
		log.Printf("[%s] taken at %04d-%02d-%02d %02d:%02d:%02d", v.Videos[i].Filename, v.Videos[i].From_year, v.Videos[i].From_month, v.Videos[i].From_day, v.Videos[i].From_hour, v.Videos[i].From_min, v.Videos[i].From_sec)
		if _, err := os.Stat(tofile); errors.Is(err, os.ErrNotExist) {
			hiklib.HikSaveFile(u, v.Videos[i].Filename, tofile)
		}
	}
}
